```js
// ===== 2048 游戏 =====
let gGrid = Array(16).fill(0);
let gScore = 0;
let gBest = parseInt(localStorage.getItem("best2048")) || 0;
let gOver = false;
let gWon = false;

let gScoreEl = document.getElementById("g2048-score");
let gBestEl = document.getElementById("g2048-best");
let gTiles = document.querySelectorAll("#g2048-grid .tile");

gBestEl.textContent = gBest;

let renderGrid = () => {
  gTiles.forEach((el, i) => {
    let v = gGrid[i];
    el.textContent = v || "";
    el.className = "tile" + (v ? " n" + v : "");
    if (v === 0) {
      el.style.color = "";
    }
  });
  gScoreEl.textContent = gScore;
  if (gScore > gBest) {
    gBest = gScore;
    gBestEl.textContent = gBest;
    localStorage.setItem("best2048", gBest);
  }
};

let addTile = () => {
  let empty = gGrid.reduce((a, v, i) => { if (v === 0) a.push(i); return a; }, []);
  if (empty.length === 0) return;
  let i = empty[Math.floor(Math.random() * empty.length)];
  gGrid[i] = Math.random() < 0.9 ? 2 : 4;
};

let initGame = () => {
  gGrid = Array(16).fill(0);
  gScore = 0;
  gOver = false;
  gWon = false;
  addTile();
  addTile();
  renderGrid();
};
```

### Task 4: 实现移动与合并逻辑

**Files:**
- Modify: `index.html` — 在上一步代码之后追加

- [ ] **Step 1: 追加核心移动函数**

```js
let slideRow = (row) => {
  let arr = row.filter(v => v);           // 去零
  for (let i = 0; i < arr.length - 1; i++) {
    if (arr[i] === arr[i + 1]) {
      arr[i] *= 2;
      gScore += arr[i];
      arr[i + 1] = 0;
    }
  }
  let merged = (arr[i] === 2048);
  arr = arr.filter(v => v);              // 再去零
  while (arr.length < 4) arr.push(0);    // 补齐
  if (merged && !gWon) {
    gWon = true;
    setTimeout(() => alert("恭喜！你达到了 2048！可以继续玩。"), 100);
  }
  return arr;
};

let getRow = (g, r) => [g[r*4], g[r*4+1], g[r*4+2], g[r*4+3]];
let setRow = (g, r, row) => { g[r*4]=row[0]; g[r*4+1]=row[1]; g[r*4+2]=row[2]; g[r*4+3]=row[3]; };
let getCol = (g, c) => [g[c], g[c+4], g[c+8], g[c+12]];
let setCol = (g, c, col) => { g[c]=col[0]; g[c+4]=col[1]; g[c+8]=col[2]; g[c+12]=col[3]; };

let arraysEqual = (a, b) => {
  for (let i = 0; i < 16; i++) { if (a[i] !== b[i]) return false; }
  return true;
};

let move = (dir) => {
  // dir: 0=左 1=右 2=上 3=下
  if (gOver) return;
  let before = gGrid.slice();

  if (dir === 0) {        // 左
    for (let r = 0; r < 4; r++) setRow(gGrid, r, slideRow(getRow(gGrid, r)));
  } else if (dir === 1) { // 右：翻转 → 左移 → 翻转
    for (let r = 0; r < 4; r++) setRow(gGrid, r, slideRow(getRow(gGrid, r).reverse()).reverse());
  } else if (dir === 2) { // 上
    for (let c = 0; c < 4; c++) setCol(gGrid, c, slideRow(getCol(gGrid, c)));
  } else if (dir === 3) { // 下：翻转 → 左移 → 翻转
    for (let c = 0; c < 4; c++) setCol(gGrid, c, slideRow(getCol(gGrid, c).reverse()).reverse());
  }

  if (!arraysEqual(before, gGrid)) {
    addTile();
    renderGrid();
    checkGameOver();
  }
};
```

```js
let checkGameOver = () => {
  // 有空格，还没结束
  if (gGrid.includes(0)) return;
  // 横向检查
  for (let r = 0; r < 4; r++) {
    for (let c = 0; c < 3; c++) {
      if (gGrid[r * 4 + c] === gGrid[r * 4 + c + 1]) return;
    }
  }
  // 纵向检查
  for (let c = 0; c < 4; c++) {
    for (let r = 0; r < 3; r++) {
      if (gGrid[r * 4 + c] === gGrid[(r + 1) * 4 + c]) return;
    }
  }
  gOver = true;
  if (!gWon) {
    setTimeout(() => alert("游戏结束！得分：" + gScore), 100);
  }
};
```

```js
let touchStartX = 0, touchStartY = 0;
document.addEventListener("touchstart", (e) => {
  touchStartX = e.touches[0].clientX;
  touchStartY = e.touches[0].clientY;
}, { passive: true });

document.addEventListener("touchend", (e) => {
  let dx = e.changedTouches[0].clientX - touchStartX;
  let dy = e.changedTouches[0].clientY - touchStartY;
  if (Math.abs(dx) < 20 && Math.abs(dy) < 20) return;

  if (Math.abs(dx) > Math.abs(dy)) {
    move(dx > 0 ? 1 : 0);
  } else {
    move(dy > 0 ? 3 : 2);
  }
});
```
